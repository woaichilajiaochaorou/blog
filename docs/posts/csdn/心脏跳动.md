# 心脏跳动

#include `&lt;stdio.h&gt;`
    #include `&lt;windows.h&gt;`
    #include `&lt;unistd.h&gt;`
    int main() {
    	float a;
    	int i; 
    	a=0.1;
    	while(1){
    		i++;
    		if(i%2==0){
    			a=0.09;
    		}else{
    			a=0.1;
    		}
    		sleep(1);
    		system("cls");
    		
        for (float y = 1.5f; y > -1.5f; y -= a ) {
            for (float x = -1.5f; x < 1.5f; x += 0.05f) {
                float a = x * x + y * y - 1;
                putchar(a * a * a - x * x * y * y * y <= 0.0f ? '*' : ' ');
            } putchar('\n');
        }
        
    }
    }
